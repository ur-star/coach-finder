import Errorbar from "../components/Errorbar.vue";

  components: {
    Errorbar,
  },
  data() {
    return {
      firstName: "",
      lastName: "",
      email: "",
      password: "",
      rePassword: "",
      showPass: false,
      valid: false,
      error: "",
      image: null,
      source: "",
      uid:""
    };
  },
  methods: {
    async signUp() {
      this.error = "";
      this.$refs.form.validate();
      if (this.valid) {
        if (this.password !== this.rePassword) {
          window.alert("Password must match");
        } else {
          try {
        let res =  await createUserWithEmailAndPassword(
              auth,
              this.email,
              this.password
            );
            const uid = res.user.uid;
            log uid

            if (this.image != null) {
              const imagesRef = ref(
                storage,
                `profile/${uid}/${this.image.name}`
              );
              await uploadBytes(imagesRef, this.image);

              this.source = await getDownloadURL(
                ref(storage, `profile/${this.uid}/${this.image.name}`)
              );
            }

            await updateProfile(auth.currentUser, {
              displayName: this.fullname,
              photoURL:this.source,
              email:this.email,
            })
              
                this.$router.push("/");
              
          } catch (error) {
            if (err.code == "auth/email-already-in-use")
              this.error = "This email already exist";
          }
        }
      } else {
        console.error("fill form correctly");
      }
    },
  },
  beforeMount() {
    this.$store.commit("drawerController", false);
  },
  computed: {
    fullname() {
      return this.firstName + " " + this.lastName;
    },
   
  },
